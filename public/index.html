export default { async fetch(request, env) { const url = new URL(request.url); const auth = "Basic " + btoa(env.DID_API_KEY); if (url.pathname === "/health") { return new Response(JSON.stringify({ ok: true, system: "CONSIA AVATAR LIVE" }), { headers: { "Content-Type": "application/json" } }); } if (url.pathname === "/v1/avatar/create") { const body = await request.json(); const res = await fetch("https://api.d-id.com/talks", { method: "POST", headers: { "Authorization": auth, "Content-Type": "application/json" }, body: JSON.stringify({ script: { type: "text", input: body.text || "CONSIA activo" }, source_url: body.avatar || "https://create-images-results.d-id.com/DefaultPresenters/Noelle_female.png" }) }); const data = await res.json(); return new Response(JSON.stringify(data), { headers: { "Content-Type": "application/json" } }); } if (url.pathname === "/v1/avatar/status") { const id = url.searchParams.get("id"); const res = await fetch(`https://api.d-id.com/talks/${id}`, { headers: { "Authorization": auth } }); const data = await res.json(); return new Response(JSON.stringify(data), { headers: { "Content-Type": "application/json" } }); } return new Response("CONSIA AVATAR ONLINE"); } };
